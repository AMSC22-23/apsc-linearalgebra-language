180  // this is the simplest case. The matrix has all column

182  this->local_product = this->local_matrix * x;
183  } else {
184  // This case is much trickier. The local matrix has fewer columns than

189  auto startcol = displacements[mpi_rank] / global_num_rows;
190  auto endcol =
