287  void update_kth_column(Scalar* new_values, int k, int* row_indices,
288  int size) {
289  CSC newA;
290  newA.m = m;
291  newA.n = n;
292 
293  // Compute the new number of nonzeros
294  int deltaNonzeros = 0;
295  for (int i = 0; i < size; i++) {
296  if (new_values[i] != 0.0) {
297  deltaNonzeros++;
298  }
299  }
300  deltaNonzeros -= offset[k + 1] - offset[k];
301 
302  // set the new number of nonzeros
303  newA.non_zeros = non_zeros + deltaNonzeros;
304 
305  // Malloc space
306  newA.offset = (int*)malloc(sizeof(int) * (n + 1));
307  newA.values = (Scalar*)malloc(sizeof(Scalar) * newA.non_zeros);
308  newA.flat_row_index = (int*)malloc(sizeof(int) * newA.non_zeros);
309 
310  // Copy the offset values before k
311  for (int i = 0; i < k + 1; i++) {
312  newA.offset[i] = offset[i];
313  }
314 
315  // Compute the new offset values for k and onwards
316  for (int i = k + 1; i < n + 1; i++) {
317  newA.offset[i] = offset[i] + deltaNonzeros;
318  }
319 
320  // Copy the old values and flat_row_index values before k
321  for (int i = 0; i < offset[k] + 1; i++) {
322  newA.values[i] = values[i];
323  newA.flat_row_index[i] = flat_row_index[i];
324  }
325 
326  // insert the new values into the values and flat_row_index from k
327  int index = 0;
328  for (int i = 0; i < size; i++) {
329  if (new_values[i] != 0.0) {
330  newA.values[offset[k] + index] = new_values[i];
331  newA.flat_row_index[offset[k] + index] = row_indices[i];
332  index++;
333  }
334  }
335 
336  // Copy the old values and flat_row_index values after k
337  for (int i = newA.offset[k + 1]; i < newA.non_zeros; i++) {
338  newA.values[i] = values[i - deltaNonzeros];
339  newA.flat_row_index[i] = flat_row_index[i - deltaNonzeros];
340  }
341 
342  // swap old with new memory
343  destoy();
344  this->m = newA.m;
345  this->n = newA.n;
346  this->initialised = 1;
347  this->offset = newA.offset;
348  this->non_zeros = newA.non_zeros;
349  this->flat_row_index = newA.flat_row_index;
350  this->values = newA.values;
351  }
