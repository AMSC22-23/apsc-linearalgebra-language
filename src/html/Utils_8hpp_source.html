34 template <typename Mat, typename Scalar>
35 void default_spd_fill(Mat &m) {
36  ASSERT((m.rows() == m.cols()), "The matrix must be squared!");
37  const Scalar diagonal_value = static_cast<Scalar>(2.0);
38  const Scalar upper_diagonal_value = static_cast<Scalar>(-1.0);
39  const Scalar lower_diagonal_value = static_cast<Scalar>(-1.0);
40  const auto size = m.rows();
41 
42  for (unsigned i = 0; i < size; ++i) {
43  m(i, i) = diagonal_value;
44  if (i > 0) {
45  m(i, i - 1) = upper_diagonal_value;
46  }
47  if (i < size - 1) {
48  m(i, i + 1) = lower_diagonal_value;
49  }
50  }
51 }
