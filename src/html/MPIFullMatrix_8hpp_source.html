106  void product(InputVectorType const &x) {
107  using namespace apsc;
108  if constexpr (ORDER_TYPE == ORDERINGTYPE::ROWWISE) {
109  // this is the simplest case. The matrix has all column
110  // (but not all rows!)
111  this->localProduct = this->localMatrix * x;
112  } else {
113  // This case is much trickier. The local matrix has fewer columns than
114  // the rows of x,
115  // so I need to reduce the global vector in input x
116  // I need to get the portion of x corresponding to the columns in the
117  // global matrix
118  auto startcol = displacements[mpi_rank] / global_nRows;
119  auto endcol = (displacements[mpi_rank] + counts[mpi_rank]) / global_nRows;
120 
121  // I copy the relevant portion of the global vector in a local vector
122  // exploiting a constructor that takes a range.
123  LinearAlgebra::Vector<Scalar> y(x.data() + startcol, x.data() + endcol);
124

126  // to a template type, where an Eigen vector can be specified

128  auto mapped_y = Eigen::Map<Eigen::VectorXd>(y.data(), y.size());
